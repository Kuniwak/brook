Sample Code

you can write the application using MVC design pattern easily,
like this: 
Namespace('sample.widget')
.use('brook *')
.use('brook.util *')
.use('brook.channel *')
.use('sample.model getModel')
.use('sample.view getView')
.define(function(ns){

var model = ns.getModel();
var view  = ns.getView();
var observeChannel = ns.observeChannel;
var observeEvent = function(element,eventName,promise){
    element.addEventListener(eventName,function(evt){promise.run(evt)},false );
};
var updateHTML = function(element){
    return ns.promise(function(n,v){
        element.innerHTML = v; n();
    });
};
ns.provide({
    registerElement:function(element){

        // handle user action and notify to the model
        // View to Model
        observeEvent( element, 'click',viaEvent.bind( model.notify('update'));

        // observing global channel ,when receiving some messages , notify to the model
        // View to Model
        observeChannel( 'shake',viaChannel.bind( model.notify('update') ));

        // observing model updates
        // Model to View
        model.observe( 'update',
            view
                .bind( updateHTML(element) )
                .bind( ns.bindAllWidget )
        );
    }
});

});
