  7 /**
  8 @name brook.model
  9 @namespace mvcにおけるmodelインタフェースを提供します。
 10 */
 11 Namespace('brook.model')
 12 .use('brook promise')
 13 .use('brook.util *')
 14 .use('brook.channel *')
 15 .use('brook.lamda *')
 16 .define(function(ns){
 17     /**
 18      * @class brook.model.createModelで生成されるインスタンスのインナークラス
 19      * @name _Model
 20      * @memberOf brook.model
 21      * @description
 22      * mvcにおけるmodelインタフェースをもったクラス
 23      */
 24     var Model = function(obj){
 25         this.methods = {};
 26         this.channels= {};
 27         for( var prop in obj ){
 28             if( !obj.hasOwnProperty(prop) )
 29                 continue;
 30             this.addMethod( prop,obj[prop]);
 31         }
 32     };
 33     Model.prototype.addMethod = function(method,promise){
 34         if( this.methods[method] )
 35             throw('already '+ method +' defined');
 36         var channel = ns.createChannel();
 37         this.methods[method] = promise.bind( channel.send() );
 38         this.channels[method] = channel;
 39         return this;
 40     };
 41     Model.prototype.notify = function(method){
 42         return ns.promise().bind( this.methods[method] );
 43     };
 44     Model.prototype.method   = function(method){
 45         if( !this.channels[method] )
 46             throw('do not observe undefined method');
 47         return this.channels[method];
 48     };
 49     var createModel = function(){
 50         return new Model;
 51     };
 52     ns.provide({
 53         createModel : createModel
 54     });
 55 });
