1 Namespace('brook.model')
  2 .use('brook promise')
  3 .use('brook.util *')
  4 .use('brook.channel *')
  5 .use('brook.lamda *')
  6 .define(function(ns){
  7     var Model = function(obj){
  8         this.methods = {};
  9         this.channels= {};
 10         for( var prop in obj ){
 11             if( !obj.hasOwnProperty(prop) )
 12                 continue;
 13             this.addMethod( prop,obj[prop]);
 14         }
 15     };
 16     Model.prototype.addMethod = function(method,promise){
 17         if( this.methods[method] )
 18             throw('already '+ method +' defined');
 19         var channel = ns.createChannel();
 20         this.methods[method] = promise.bind( channel.send() );
 21         this.channels[method] = channel;
 22         return this;
 23     };
 24     Model.prototype.notify = function(method){
 25         return ns.promise().bind( this.methods[method] );
 26     };
 27     Model.prototype.method   = function(method){
 28         if( !this.channels[method] )
 29             throw('do not observe undefined method');
 30         return this.channels[method];
 31     };
 32     var createModel = function(){
 33         return new Model;
 34     };
 35     ns.provide({
 36         createModel : createModel
 37     });
 38 });
