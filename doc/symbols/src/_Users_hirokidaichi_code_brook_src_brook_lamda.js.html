 12 Namespace('brook.lamda')
 13 .define(function(ns){
 14     var cache = {};
 15     var hasArg = function(expression){
 16         return expression.indexOf('->') >= 0;
 17     };
 18     var parseExpression = function(expression){
 19         var fixed = hasArg( expression ) ? expression : "$->"+expression;
 20         var splitted = fixed.split("->");
 21         var argsExp = splitted.shift();
 22         var bodyExp = splitted.join('->');
 23         return {
 24             argumentNames : argsExp.split(','),
 25             body   : hasArg(bodyExp) ? lamda( bodyExp ).toString() : bodyExp
 26         };
 27     };
 28     var lamda = function(expression){
 29         if( cache[expression] )
 30             return cache[expression];
 31         var parsed = parseExpression(expression);
 32         var func = new Function( parsed.argumentNames,"return ("+ parsed.body + ");");
 33         cache[expression] = func;
 34         return func;
 35     };
 36     ns.provide({
 37         lamda : lamda
 38     });
 39 });
