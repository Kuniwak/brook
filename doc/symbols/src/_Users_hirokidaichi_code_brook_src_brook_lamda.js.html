  7 /**
  8 @name brook.lamda
  9 @namespace 簡単に小さな関数を作る為のテンプレートを提供します。
 10 */
 11 
 12 Namespace('brook.lamda')
 13 .define(function(ns){
 14     var cache = {};
 15     var hasArg = function(expression){
 16         return expression.indexOf('->') >= 0;
 17     };
 18     var parseExpression = function(expression){
 19         var fixed = hasArg( expression ) ? expression : "$->"+expression;
 20         var splitted = fixed.split("->");
 21         var argsExp = splitted.shift();
 22         var bodyExp = splitted.join('->');
 23         return {
 24             argumentNames : argsExp.split(','),
 25             body   : hasArg(bodyExp) ? lamda( bodyExp ).toString() : bodyExp
 26         };
 27     };
 28     /**
 29      * @name lamda
 30      * @function
 31      * @memberOf brook.lamda
 32      * @param {string} expression
 33      * @return {function}
 34      * @description
 35      * 文字列表現を受け取り、シンプルな関数を生成します。
 36      * @example
 37      * var f = lamda('$ * $'); // 第一引数を二乗する関数
 38      * @example
 39      * var f = lamda('x,y-> x + y'); // xとyを受け取って、x+yを返す
 40      * @example
 41      * var f = lamnda('x->y->z-> x+y+z'); // 部分適用できる関数を作る
 42      */
 43     var lamda = function(expression){
 44         if( cache[expression] )
 45             return cache[expression];
 46         var parsed = parseExpression(expression);
 47         var func = new Function( parsed.argumentNames,"return ("+ parsed.body + ");");
 48         cache[expression] = func;
 49         return func;
 50     };
 51     ns.provide({
 52         lamda : lamda
 53     });
 54 });
