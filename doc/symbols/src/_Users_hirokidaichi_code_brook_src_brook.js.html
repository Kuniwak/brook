  7 /**
  8 @name brook
  9 @namespace brookライブラリ群のルートとなる名前空間です。promiseの生成処理を持っています。
 10 */
 11 Namespace('brook').define(function(ns){
 12     var VERSION = "0.01";
 13     /**
 14      * @class brook.promiseで生成されるインスタンスのインナークラス
 15      * @name _Promise
 16      * @memberOf brook
 17      * @description
 18      * 実行する前の次の処理をもつオブジェクトです。
 19      * Promiseインスタンスはbind関数で結合する事が出来ます。連続した非同期/同期の処理をデータの流れとして抽象化して結合する事が出来ます。
 20      * subscribe/forEach/runなどの処理を実行するまでは、結合した処理は実行される事はありません。
 21      * また、コンストラクタは公開されていません。brook.promiseがファクトリとなっています。
 22      */
 23     var Promise = function(next){
 24         this.next = next ||  function(next,val){ return next(val); };
 25     };
 26     (function(proto){
 27     /**#@+
 28      * @methodOf brook._Promise.prototype
 29      */
 30 
 31     /**
 32      * @name concat
 33      * @param {Promise} promise
 34      */
 35     proto.concat = function(promise){
 36         var _before = this;
 37         var after  = promise;
 38         var next   = function(n,val){
 39             return _before.subscribe( promise.ready(n),val);
 40         };
 41         return new Promise(next);
 42     };
 43     /**
 44      * @name bind
 45      * @param {Promise} promise
 46      */
 47     proto.bind = function(){
 48         var r = this;
 49         for( var i = 0,l = arguments.length;i<l;i++){
 50             var s = arguments[i];
 51             s = ( s instanceof Promise) ? s : promise( s );
 52             r = r.concat( s );
 53         }
 54         return r;
 55     };
 56     /**
 57      * @name ready
 58      * @param {Promise} promise
 59      */
 60     proto.ready = function(n){
 61         var proc = this.next;
 62         return function(val){
 63             return proc(n,val);
 64         };
 65     };
 66     /**
 67      * @name run
 68      * @param {Promise} promise
 69      */
 70     proto.run = function(val){
 71         this.subscribe( undefined , val );
 72     };
 73     /**
 74      * @name subscribe
 75      * @param {Promise} promise
 76      */
 77     proto.subscribe = function(next,val){
 78         var next = next ? next : function(){};
 79         if( !this.errorHandler )
 80             return this.next(next,val);
 81         
 82         try {
 83             this.next(next,val);
 84         }
 85         catch(e){
 86             this.onError(e);
 87         }
 88     };
 89     /**
 90      * @name forEach
 91      * @param {Promise} promise
 92      */
 93     proto.forEach = proto.subscribe;
 94     /**
 95      * @name setErrorHandler
 96      * @param {Promise} promise
 97      */
 98     proto.setErrorHandler = function(promise){
 99         this.errorHandler = promise;
100     };
101     /**
102      * @name onError
103      */
104     proto.onError = function(e){
105         (this.errorHandler||new Promise).subscribe(function(){},e);
106     };
107     /**#@-*/
108     })(Promise.prototype);
109     /**
110      * @name promise
111      * @function
112      * @memberOf brook
113      * @param {function} next
114      * @return {Promise}
115      * @description
116      * プロミスを生成するファクトリメソッドです。nextは、さらに次の処理を第一引数に受け取り、第二引数に前回の処理の結果を受け取ります。
117      * 引数が無い場合は、データをそのまま次の処理に送るpromiseを生成します。
118      * @example
119      * var p = ns.promise(function(next,value){ next(value+1)});
120      * @example
121      * var p = ns.promise();
122      */
123     var promise = function(next){return new Promise(next)};
124     ns.provide({
125         promise : promise,
126         VERSION : VERSION
127     });
128 });
