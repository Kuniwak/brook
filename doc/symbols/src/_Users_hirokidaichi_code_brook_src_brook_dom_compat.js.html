 25             return string.replace(/-+(.)?/g, function(match, chr) {

 27             });
 28         };

 40     var ClassList = function(element){
 41         this._element = element;
 42         this._refresh();
 43     };
 44     var classList = function(element){
 45         return new ClassList(element);
 46     };
 47 
 48     ClassList.prototype = new Array;
 49     (function(proto){
 50         var check = function(token) {
 51             if (token == "") {
 52                 throw "SYNTAX_ERR";
 53             }
 54             if (token.indexOf(/\s/) != -1) {
 55                 throw "INVALID_CHARACTER_ERR";
 56             }
 57         };
 58         this._fake = true;
 59         this._refresh = function () {
 60             var clss = this._element.getAttribute("class");
 61             if (!clss) {
 62                 return this;
 63             }
 64             var classes = clss.split(/\s+/);
 65             if (classes.length && classes[0] == "") {
 66                 classes.shift();
 67             }
 68             if (classes.length && classes[classes.length - 1] == "") {
 69                 classes.pop();
 70             }
 71             this.length = classes.length;
 72             if (this.length == 0) {
 73                 return this;
 74             }
 75             for (var i = 0; i < this.length; ++i) {
 76                 this[i] = classes[i];
 77             }
 78             return this;
 79         };
 80         this.item = function (i) {
 81             return this[i] || null;
 82         }
 83         this.contains = function (token) {
 84             check(token);
 85             for (var i = 0; i < this.length; ++i) {
 86                 if (this[i] == token) {
 87                     return true;
 88                 }
 89             }
 90             return false;
 91         }
 92         this.add = function (token) {
 93             check(token);
 94             for (var i = 0; i < this.length; ++i) {
 95                 if (this[i] == token) {
 96                     return;
 97                 }
 98             }
 99             this.push(token);
100             this._element.setAttribute("class", this.join(" "));
101         }
102         this.remove = function (token) {
103             check(token);
104             for (var i = 0; i < this.length; ++i) {
105                 if (this[i] == token) {
106                     this.splice(i, 1);
107                     this._element.setAttribute("class", this.join(" "));
108                 }
109             }
110         }
111         this.toggle = function (token) {
112             check(token);
113             for (var i = 0; i < this.length; ++i) {
114                 if (this[i] == token) {
115                     this.remove(token);
116                     return false;
117                 }
118             }
119             this.add(token);
120             return true;
121         }
122     }).apply(ClassList.prototype);

124     var hasClassName = function(element,className){
125         var classSyntax = element.className;
126         if ( !(classSyntax && className) ) return false;
127         return (new RegExp("(^|\\s)" + className + "(\\s|$)").test(classSyntax)); 
128     };

130         if( document.getElementsByClassName ) return document.getElementsByClassName( className );
131         var allElements = document.getElementsByTagName('*');
132         var ret = [];
133         for(var i=0,l=allElements.length;i<l;i++){
134             if( !hasClassName( allElements[i] , className ) )
135                 continue;
136             ret.push( allElements[i] )
137         }
138         return ret;
