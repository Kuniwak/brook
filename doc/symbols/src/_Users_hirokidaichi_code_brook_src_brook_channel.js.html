  7 /**
  8 @name brook.channel
  9 @namespace promiseをベースとしたobserver patternのシンプルな実装を提供します。
 10 */
 11 Namespace('brook.channel')
 12 .use('brook promise')
 13 .define(function(ns){
 14     /**
 15      * @class brook.channel.createChannelで生成されるインスタンスのインナークラス
 16      * @name _Channel
 17      * @memberOf brook.channel
 18      * @description
 19      * promiseを登録できるobserverクラス
 20      */
 21     var Channel = function(){
 22         this.queue = [];
 23         this.promises = [];
 24     };
 25     (function(proto){
 26     /**#@+
 27      * @methodOf brook.channel._Channel.prototype
 28      */
 29         var through = function(k){return k};
 30         proto.sendMessage = function(msg){
 31             this.queue.push(msg);
 32             while( this.queue.length ){
 33                 var v = this.queue.shift();
 34                 for( var i = 0,l= this.promises.length;i<l;i++){
 35                     this.promises[i].run(v);
 36                 }
 37             }
 38         };
 39         /**
 40          * @name send
 41          */
 42         proto.send = function(func){
 43             var func = ( func ) ? func : through;
 44             var _self = this;
 45             return ns.promise(function(next,val){
 46                 _self.sendMessage(func(val));
 47                 next(val);
 48             });
 49         };
 50         /**
 51          * @name observe
 52          */
 53         proto.observe = function(promise){
 54             this.promises.push(promise);
 55         };
 56     /**#@-*/
 57     })(Channel.prototype);
 58 
 59     var channel = function(name){
 60         if( name )
 61             return getNamedChannel(name);
 62         return new Channel;
 63     };
 64 
 65     var NAMED_CHANNEL = {};
 66     var getNamedChannel = function(name){
 67         if( NAMED_CHANNEL[name] )
 68             return NAMED_CHANNEL[name];
 69         NAMED_CHANNEL[name] = new Channel;
 70         return NAMED_CHANNEL[name];
 71     };
 72     var observeChannel = function(name,promise){
 73         getNamedChannel( name ).observe( promise );
 74     };
 75     var sendChannel = function(name,func){
 76         var channel = getNamedChannel( name );
 77         return channel.send(func);
 78     };
 79     ns.provide({
 80         channel        : channel,
 81         sendChannel    : sendChannel,
 82         observeChannel : observeChannel,
 83         createChannel  : function(){ return new Channel;}
 84     });
 85 });
