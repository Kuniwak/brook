1 Namespace('brook.widget')
  2 .use('brook promise')
  3 .use('brook.channel *')
  4 .use('brook.util *')
  5 .use('brook.dom.compat *')
  6 .define(function(ns){
  7     var TARGET_CLASS_NAME = 'widget';
  8 
  9     var classList = ns.classList;
 10     var dataset   = ns.dataset;
 11     var channel   = ns.channel;
 12 
 13     var removeClassName = function(className,element){
 14         classList(element).remove(className);
 15     };
 16     var elementsByClassName = ns.promise(function(n,v){
 17         v = v || TARGET_CLASS_NAME;
 18         n([v,Array.prototype.slice.call(ns.getElementsByClassName(v))]);
 19     });
 20     var mapByNamespace = ns.promise(function(n,val){
 21         var targetClassName = val[0];
 22         var widgetElements  = val[1];
 23         var map = {};
 24         for( var i = 0,l = widgetElements.length;i<l;i++){
 25             var widget = widgetElements[i];
 26             removeClassName((targetClassName||TARGET_CLASS_NAME),widget);
 27             var dataset = ns.dataset(widget);
 28             if( !dataset.widgetNamespace ) continue;
 29             if( !map[dataset.widgetNamespace] ) map[dataset.widgetNamespace] = [];
 30             map[dataset.widgetNamespace].push( widget );
 31         }
 32         n(map);
 33     });
 34     var registerElements = ns.promise(function(n,map){
 35         for( var namespace in map ){
 36             if( !map.hasOwnProperty( namespace ) ) continue;
 37             var targets = map[namespace];
 38             Namespace.use([namespace , '*'].join(' ')).apply(function(_ns){
 39                 if (_ns.registerElement) {
 40                     for( var i = 0,l=targets.length;i<l;i++){
 41                         _ns.registerElement(targets[i]);
 42                     }
 43                 } else if (_ns.registerElements) {
 44                     _ns.registerElements( targets );
 45                 } else {
 46                     throw('registerElement or registerElements not defined in ' + namespace);
 47                 }
 48             });
 49         }
 50     });
 51 
 52     var bindAllWidget = ns.sendChannel('widget');
 53 
 54     var updater  = ns.from( channel('widget') )
 55         .bind( ns.lock('class-seek') )
 56         .bind( elementsByClassName )
 57         .bind( mapByNamespace )
 58         .bind( ns.unlock('class-seek') )
 59         .bind( registerElements );
 60   
 61     updater.subscribe();
 62     ns.provide({
 63         bindAllWidget : bindAllWidget
 64     });
 65 });
